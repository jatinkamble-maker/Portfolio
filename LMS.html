    // SIMPLE DELETE METHOD (Using Index)
    public void removeBook(int index) {
        if (index < 0 || index >= bookNames.length) {
            System.out.println("Invalid number!");
            return;
        }

        // Create smaller arrays (size - 1)
        String[] nb = new String[bookNames.length - 1];
        String[] na = new String[authors.length - 1];
        Integer[] ni = new Integer[bookIds.length - 1];
        String[] ns = new String[isbn.length - 1];

        int k = 0; // counter for new arrays
        for (int i = 0; i < bookNames.length; i++) {
            if (i == index) {
                continue; // Skip the one we want to delete
            }
            nb[k] = bookNames[i];
            na[k] = authors[i];
            ni[k] = bookIds[i];
            ns[k] = isbn[i];
            k++;
        }

        // Update the main arrays
        bookNames = nb;
        authors = na;
        bookIds = ni;
        isbn = ns;

        saveToFile();
    }

    // LOAD BOOKS FROM TEXT FILE
    public void loadBooks() {
        try {
            File f = new File(bookFile);
            Scanner reader = new Scanner(f);
            while (reader.hasNextLine()) {
                String line = reader.nextLine();
                String[] parts = line.split(",");
                if (parts.length == 4) {
                    addBookForLoading(parts[0], parts[1], Integer.parseInt(parts[2]), parts[3]);
                }
            }
            reader.close();
        } catch (Exception e) {
            System.out.println("Could not load books.");
        }
    }
