AI Traffic Monitoring System – Python Code


import cv2
import pytesseract
import math
import csv
from collections import defaultdict, deque
from datetime import datetime

# ---------------- CONFIGURATION ---------------- #
VIDEO_PATH = "hi.mp4"
SPEED_LIMIT = 60                  # km/h
PIXEL_TO_METER = 0.05             # calibration factor
SPEED_WINDOW = 5                  # smoothing window
STOP_LINE_Y = 300                 # red signal stop line
ALLOWED_DIRECTION = (1, 0)        # allowed motion: left → right

# If Tesseract path is needed (macOS Intel / Apple Silicon example)
# pytesseract.pytesseract.tesseract_cmd = "/opt/homebrew/bin/tesseract"

# ---------------- DATABASES ---------------- #
vehicle_history = defaultdict(list)
violation_count = defaultdict(int)
rfid_status = defaultdict(lambda: "ACTIVE")
speed_buffer = deque(maxlen=SPEED_WINDOW)

# ---------------- VIDEO LOAD ---------------- #
cap = cv2.VideoCapture(VIDEO_PATH)
if not cap.isOpened():
    raise IOError("❌ Cannot open video file")

FPS = cap.get(cv2.CAP_PROP_FPS)
DT = 1 / FPS if FPS > 0 else 0.1

# ---------------- NUMBER PLATE DETECTOR ---------------- #
plate_cascade = cv2.CascadeClassifier(
    cv2.data.haarcascades + "haarcascade_russian_plate_number.xml"
)

# ---------------- FUNCTIONS ---------------- #
def compute_speed(p1, p2):
    dx, dy = p2[0] - p1[0], p2[1] - p1[1]
    distance_pixels = math.sqrt(dx**2 + dy**2)
    speed_mps = (distance_pixels * PIXEL_TO_METER) / DT
    return speed_mps * 3.6

def is_wrong_route(prev, curr):
    dx, dy = curr[0] - prev[0], curr[1] - prev[1]
    dot = dx * ALLOWED_DIRECTION[0] + dy * ALLOWED_DIRECTION[1]
    return dot < 0

def detect_plate(frame):
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    plates = plate_cascade.detectMultiScale(gray, 1.1, 4)

    for (x, y, w, h) in plates:
        plate_img = gray[y:y+h, x:x+w]
        text = pytesseract.image_to_string(
            plate_img, config="--psm 7"
        ).strip()
        if text:
            return text, (x, y, w, h)

    return "UNKNOWN", None

def log_violation(vehicle_id, violations, speed):
    with open("violations.csv", "a", newline="") as f:
        writer = csv.writer(f)
        writer.writerow([
            vehicle_id,
            ",".join(violations),
            round(speed, 2),
            rfid_status[vehicle_id],
            datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        ])

# ---------------- MAIN LOOP ---------------- #
prev_center = None
vehicle_id = "UNKNOWN"

print("\n🚦 AI Traffic Monitoring Started\n")

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        print("\n🎬 Video completed. Exiting system...")
        break

    h, w, _ = frame.shape

    # -------- Simulated vehicle tracking (replaceable with YOLO) -------- #
    center = (w // 2 + len(vehicle_history[vehicle_id]) * 2, h // 2)

    # -------- Number Plate Detection -------- #
    plate_text, plate_box = detect_plate(frame)
    if plate_text != "UNKNOWN":
        vehicle_id = plate_text

    vehicle_history[vehicle_id].append(center)

    if prev_center is not None:
        speed = compute_speed(prev_center, center)
        speed_buffer.append(speed)
        avg_speed = sum(speed_buffer) / len(speed_buffer)

        violations = []

        # Overspeeding
        if avg_speed > SPEED_LIMIT:
            violations.append("Overspeeding")

        # Signal Jumping
        if prev_center[1] < STOP_LINE_Y <= center[1]:
            violations.append("Signal Jumping")

        # Wrong Route
        if is_wrong_route(prev_center, center):
            violations.append("Wrong Route")

        if violations:
            violation_count[vehicle_id] += 1

            if violation_count[vehicle_id] >= 3:
                rfid_status[vehicle_id] = "BLOCKED"

            log_violation(vehicle_id, violations, avg_speed)
